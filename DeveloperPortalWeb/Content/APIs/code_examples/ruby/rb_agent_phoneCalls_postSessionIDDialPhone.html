# Calls the API to dial an outbound call.
#
# * *Arguments* :
#  - +token+ -> a token hash
#  - +sessionID+ -> a string
#  - +phoneNumber+ -> a string
#  - +skillName+ -> a string
#  - +ParentContactId+ -> a int
#
# * *Returns* :
#   - an HTTP response
#
# * *Raises* :
#   - +ArgumentError+ -> if the token object doesn't have expected fields
#   - +ArgumentError+ -> if sessionID isn't a nonempty string
#   - +ArgumentError+ -> if phoneNumber isn't a string
#   - +ArgumentError+ -> if skillName isn't a string
#
def dialPhone(token, sessionID, phoneNumber="", skillName="")
  # Check that the token is a valid token.
  if (token["access_token"] == nil || 
      token["resource_server_base_uri"] == nil)
    raise ArgumentError, "The function didn't get a token it could understand"
  end
  
  # Check that sessionID is a nonempty string.
  unless sessionID.is_a?(String) && sessionID != ""
    raise ArgumentError, "sessionID must be a nonempty string"
  end
  
  # Check that phoneNumber is a valid phone number.
  unless phoneNumber.is_a?(String) &&
         phoneNumber =~ /(\(\d{3}\))?(\d{3}\-?)?\d{3}\-?\d{4}(\s?ext\.?\s?\d{4})?/
    raise ArgumentError, "phoneNumber must be a string"
  end
  
  # Check that skillName is a string.
  unless skillName.is_a? String
    raise ArgumentError, "skillName must be a string"
  end
  
  # Pull the access token and base URL from the response body.
  accessToken = token["access_token"]
  baseURL = token["resource_server_base_uri"]
  
  # Encode the session ID.
  sessionID.gsub! "=", "%3D"

  # Create the URL that accesses the API.
  apiURL = "services/{version}/agent-sessions/#{sessionID}/"\
           "dial-phone?phoneNumber=#{phoneNumber}&skillName=#{skillName}&ParentContactId=#{ParentContactId}"
  uri = URI.parse(baseURL + apiURL)

  # Create the request headers.
  headers = 
    { "Accept" => "application/json, text/javascript, */*; q=0.01",
      "Authorization" => "bearer #{accessToken}",
      "Content-Length" => "0",
      "Content-Type" => "application/x-www-form-urlencoded",
      "Data-Type" => "json" }

  # Set up an HTTP object.
  http = Net::HTTP.new uri.host, uri.port

  # Make the request an HTTPS request.
  http.use_ssl = true

  # Delete this line and be sure you have a valid certificate.
  # The default Ruby library, net/https, doesn't check the validity of a 
  # certificate during a handshake. Using VERIFY_NONE is a simple, and insecure,
  # hack to get around this limitation.
  http.verify_mode = OpenSSL::SSL::VERIFY_NONE

  # Make the request and store the response.
  response = http.post(uri.path, "", headers)
end
