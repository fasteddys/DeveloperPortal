// importing the scalaj-http library
package scalaj.http

object getCompletedContacts {

	def getCompletedContacts() {

		//Give the specified url ,accessToken

			val accessToken = "";
			val BASEURL = "";

			//Check if accessToken is empty or null
			if(accessToken!= null && !accessToken.isEmpty())
			{
				//Add all necessary headers and Make the http request
				val response = Http( BASEURL +"services/{version}/contacts/completed")
						.params("startDate"->"ISO 8601 formatted date/time",
							"endDate"->"ISO 8601 formatted date/time",
							"updatedSince"->"ISO 8601 formatted date/time",
							"mediaType"->"string",
							"fields"->"string",
							"skip"->"integer",

							"orderby"->"string",
							"skillId"->"integer",
							"campaignId"-> "integer",
							"agentId"-> "integer",
							"teamId"-> "integer",
							"toAddr"-> "string",
							"fromAddr"->"string",
							"isLogged"->"boolean",
							"isRefused"->"boolean",
							"isTakeover"->"boolean",
							"tags"->"string",
							"analyticsProcessed"->"boolean")
						.headers(("Authorization", "Bearer " + accessToken),("content-type", "application/json")).asString

				//TODO: the response comes back as a JSON document. Replace the following code with your preferred JSON //package to handle the response appropriately
				println(response);
