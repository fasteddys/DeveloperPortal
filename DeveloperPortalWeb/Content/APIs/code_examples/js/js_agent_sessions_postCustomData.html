function postCustomDataToContact() {
    var postCustomDataToContactPayload = {
        'indicatorName': 'string',
        'data': 'string'
    }
    $.ajax({
        //The baseURI variable is created by the result.resource_server_base_uri,
        //which is returned when getting a token and should be used to create the URL base
        'url': baseURI + 'services/{version}/agent-sessions/' + sessionId + '/interactions/' + contactId + '/custom-data',
        'type': 'POST',
        'headers': {
            //Use access_token previously retrieved from inContact token service
            'Authorization': 'bearer ' + accessToken,
            'content-Type': 'application/json'
        },
        'data': JSON.stringify(postCustomDataToContactPayload),
        'success': function (result, status, statusCode) {
            //Process error actions
            return result;
        },
        'error': function (XMLHttpRequest, textStatus, errorThrown) {
            //Process error actions
            return false;
        }
    })
}
