function getNextEvent(sessionId, timeout) {
    $.ajax({
        //The baseURI variable is created by the result.base_server_base_uri
        //which is returned when getting a token and should be used to create the URL Base.
        'url': baseURI + 'services/{version}/agent-sessions/' + sessionId + '/get-next-event?timeout=' + timeout,
        'type': 'GET',
        'headers': {
            //Use access_token previously retrieved from inContact token service
            'Authorization': 'bearer ' + accessToken,
            'content-Type': 'application/x-www-form-urlencoded'
        },
        'success': function (result) {
            //Process success actions
            //Note the scenarios listed below are only a few of the Events that can be returned
            for (i = 0; i < result.events.length; i++) {
                var event = result.events[i];
                //Checking for Chat Contact
                if (event.Type == 'ChatContactEvent') {
                    //Add desired functionality
                    if (event.Status == 'Incoming') {
                        //Add desired functionality
                    }
                }
                //Checking for Consult Call
                if (event.Type == 'CallContactEvent') {
                    //Add desired functionality
                    if (event.CallType == 'Consult') {
                        if (event.Status == 'Ringing') {
                            //Add desired functionality
                        }
                    }
                }
                //Checking for Work Item Contact
                if (event.Type == 'WorkItemContactEvent') {
                    if (event.Status == 'Incoming') {
                    //Add desired functionality
                    }
                }
            }
            return true;
        },
        'error': function (XMLHttpRequest, textStatus, errorThrown) {
            //Process error actions
            return false;
        },
        'complete': function (result) {
            //Process complete actions
            return true;
        }
    });
}
