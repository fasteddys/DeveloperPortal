function setDisposition(sessionId, contactId) {
    var setDispositionPayload = {
        'dispositionName': 'string',
        'dispositionNotes': 'string',
        'commitmentAmount': 'string',
        'callbackTime': 'ISO 8601 formatted date/time',
        'callbackNumber': 'string',
    }
    $.ajax({
        //The baseURI variable is created by the result.base_server_base_uri
        //which is returned when getting a token and should be used to create the URL base
        'url': baseURI + 'services/{version}/agent-sessions/' + sessionId + '/interactions/' + contactId + '/disposition',
        'type': 'PUT',
        'headers': {
            //Use access_token previously retrieved from inContact token service
            'Authorization': 'bearer ' + accessToken,
            'content-Type': 'application/json'
        },
        'data': JSON.stringify(setDispositionPayload),
        'success': function (result) {
            //Process success actions
            return result;
        },
        'error': function (XMLHttpRequest, textStatus, errorThrown) {
            //Process error actions
            return false;
        }
    });
}
