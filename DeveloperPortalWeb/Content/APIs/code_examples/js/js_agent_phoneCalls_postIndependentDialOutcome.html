function dispositionIndependentCall() {
    dispositionIndependentCallPayload = {
        'outcome': 'string - Answered Call, Busy, Fax, Intercept, No Answer'
    }
    $.ajax({
        //The baseURI variable is created by the result.resource_server_base_uri,
        //which is returned when getting a token and should be used to create the URL base
        'url': baseURI + 'services/{version}/agent-sessions/' + sessionId + '/interactions/' + contactId + '/independent-dial-outcome',
        'type': 'POST',
        'headers': {
            //Use access_token previously retrieved from inContact token service
            'Authorization': 'bearer ' + accessToken,
            'content-Type': 'application/json'
        },
        'data': JSON.stringify(dispositionIndependentCallPayload),
        'success': function (result, status, statusCode) {
            //Process success actions
            return result;
        },
        'error': function (XMLHttpRequest, textStatus, errorThrown) {
            //Process error actions
            return false;
        }
    })
}
